//Sum_of_Fractional_Sequence 1+1/2-1/3+1/4-1/5...+1/1200
{
    class Program
    {
        static void Main(string[] args)
        {
            double x = 0;
            for (int i = 2; i <= 1200; i++)
            {
                x = x + (Math.Pow(-1,i)) / i;
            }
            Console.WriteLine("Ñóììà {0:F3}",x+1);
            Console.ReadKey();
        }
    }
}
