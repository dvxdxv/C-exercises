            {
                sequence[i] = 1+i;

             }
            for (int i = 0; i < n/2; i++)

                int temp = sequence[i];
                System.Random random = new System.Random();
                int i1 = random.Next(n / 2, n);
                sequence[i] = sequence[i1];
