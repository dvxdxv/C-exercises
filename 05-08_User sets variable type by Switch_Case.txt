//05-08_User sets variable type by Switch/Case
{
    class Program
    {
        static void Main(string[] args)
        {
            var n = 0D;
            Console.WriteLine("Введите тип переменной \"n\":\nДля целых чисел нажмите - 0\nДля дробных чисел нажмите - 1\nДля текстового представления нажмите - 2");
            int type = int.Parse(Console.ReadLine());
            Console.WriteLine("Введите значение переменной \"n\":");
            var input = Console.ReadLine();
            switch (type)
            {
                case 0: Console.WriteLine((n = int.Parse(input)) + 1);
                    break;
                case 1:
                    Console.WriteLine((n = double.Parse(input)) + 1);
                    break;
                case 2:
                    Console.WriteLine(input+"*");
                    break;
                default:
                    Console.WriteLine("Вы ввели неправильный тип переменной");
                    break;
            }


        }
    }
}
